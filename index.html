// Operation

How It Works

01
Patrol
Robot moves autonomously in forward direction along designated perimeter.
02
Scan
Ultrasonic sensor continuously measures distance to detect obstacles.
03
Avoid
On detecting an obstacle <20 cm, robot backs up and turns to reroute.
04
Detect
PIR sensor triggers when unauthorized movement is sensed in the zone.
05
Alert
GSM module fires an SMS to security personnel within seconds.

Arduino Firmware

night_patrol_robot.ino 
/*  Night Patrolling Robot · Sept 2025
    PIR + Ultrasonic + L298N Motors + SIM800L GSM  */

#include <SoftwareSerial.h>
SoftwareSerial gsm(11, 12);

int PIR=2, TRIG=3, ECHO=4;
int IN1=5,IN2=6,IN3=7,IN4=8,ENA=9,ENB=10;

void setup() {
  Serial.begin(9600);
  gsm.begin(9600);
  pinMode(PIR, INPUT);
  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);
  // ... motor pins setup ...
  delay(2000);
  gsm.println("AT+CMGF=1"); // SMS text mode
}

void loop() {
  long dist = getDistance();

  if (dist > 0 && dist < 20) {
    stopCar();
    moveBackward(); delay(300);
    turnRight();    delay(300);
  } else {
    moveForward();
  }

  if (digitalRead(PIR) == HIGH) {
    sendSMS("Alert! Motion detected by patrol robot.");
    delay(5000);
  }
}

long getDistance() {
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  long d = pulseIn(ECHO, HIGH, 30000);
  return d * 0.034 / 2;
}

void sendSMS(String msg) {
  gsm.println("AT+CMGS=\"+91XXXXXXXXXX\"");
  delay(300);
  gsm.println(msg);
  gsm.write(26); // Ctrl+Z sends the SMS
}
